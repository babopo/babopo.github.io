排序算法

    • function quickSort(arr) {
    const len = arr.length
    if(len <= 1) {
        return arr
    }
    let p = len / 2 | 0
    const left = []
    const mid = []
    const right = []
    for(let i = 0; i < len; i++) {
        if(arr[i] < arr[p]) {
            left.push(arr[i])
        } else if(arr[i] === arr[p]){
            mid.push(arr[i])
        } else {
            right.push(arr[i])
        }
    }
    return [...quickSort(left), ...mid, ...quickSort(right)]
} 
//不考虑空间复杂度

    • function mergeSort(arr) {
    if(arr.length === 1) {
        //考虑一下边界即可
        return arr
    }
    const mid = arr.length >> 1
    const left = mergeSort(arr.slice(0, mid))
    const right = mergeSort(arr.slice(mid))
    return merge(left, right)

    function merge(left, right) {
        let l = 0
        let r = 0
        let res = []
        while(l < left.length && r < right.length) {
            if(left[l] < right[r]) {
                res.push(left[l++])
            }
            if(left[l] > right[r]) {
                res.push(right[r++])
            }
            if(left[l] === right[r]) {
                res.push(left[l++])
                r++
            }
        }
        if(l < left.length) {
            res = res.concat(left.slice(l))
        }
        if(r < right.length) {
            res = res.concat(right.slice(r))
        }
        return res
    }
}

节流和防抖

// 节流
const throttle = (fn, wait) => {
    let pre = Date.now()
    return function(...args) {
        let now = Date.now()
        if(now - pre > wait) {
            pre = now
            fn.apply(null, args)
        }
    }
}

// 防抖
const debounce = (fn, wait) => {
    let timer = null
    return function(...args) {
        if(timer) {
            clearTimeout(timer)
        }
        timer = setTimeout(() => {
            fn.apply(null, fn)
        }, wait)
    }
}

``` js function promise(fn) { this.state = 'pending' this.value = null this.reason = null // 存放成功和失败的then回调 this.onresolved = [] this.onrejected = [] const self = this function resolve(value) { if(self.state === 'pending') { self.value = value self.state = 'fullFilled' self.onresolved.forEach(it => it(self.value)) } } function reject(reason) { if(self.state === 'pending') { self.reason = reason self.state = 'rejected' self.onrejected.forEach(it => it(self.reason)) } } try { fn(resolve, reject) } catch(e) { reject(e) } }

promise.prototype.then = function(onfullfilled, onrejected) {
const self = this
if(self.state === ‘fullFilled’) {
onfullfilled(self.value)
}
if(self.state === ‘rejected’) {
onrejected(self.reason)
}
if(self.state === ‘pending’) {
self.onresolved.push(onfullfilled)
self.onrejected.push(onrejected)
}
return self
}

  //eventEmitter
  class eventEmitter {
      constructor() {
          this.map = {}
      }

      emit(name, ...args) {
          this.map[name].forEach(it => {
              it.apply(null, args)
          })
      }

      on(name, fn) {
          if(this.map[name]) {
              this.map[name].push(fn)
          } else {
              this.map[name] = [fn]
          }
      }

      off(name, fn) {
        //监听同一个事件多次，每次也应当只删除一个  
          if(this.map[name]) {
              const idx = this.map[name].indexOf(fn)
              if(idx > -1) {
                  this.map[name].splice(idx, 1)
              }
          }
      }
  }

实现JSON parser

  • // JSON parser
function jsonParser(str) {
    let res
    if(str[0] === '{') {
        // 对象
        const content = str.slice(1, str.length - 1)
        res = jsonObject(content)
    } else if (str[0] === '[') {        
        // 数组
        const content = str.slice(1, str.length - 1)
        res = jsonArray(content)
    } else if (str[0] === "'" || str[0] === '"') {
        // 是字符串
        res = str.slice(1, str.length - 1)
    } else if (str === 'true') {
        // 布尔值
        res = true
    } else if (str === 'false') {
        res = false
    } else if (str === 'null') {
        res = null
    } else {
        // 那就只能是数字了
        res = +str
    }
    return res 

    function jsonObject(str) {
        // 处理剥去两个外层{}的对象
        let curr = 0
        // 指针
        const res = {}
        while(curr < str.length) {
            // 找出属性名
            const propStart = curr
            while(str[curr] !== ':') {
                curr++ 
            }
            const prop = jsonParser(str.slice(propStart, curr))
            const valStart = ++curr
            let stack = 0
            while(curr < str.length && (str[curr] !== ',' || stack !== 0)) {
                if(str[curr] === '{' || str[curr] === '[') {
                    stack++
                }
                if(str[curr] === '}' || str[curr] === ']') {
                    stack--
                }
                curr++
            }
            const val = jsonParser(str.slice(valStart, curr++))
            res[prop] = val
        }
        return res
    }

    function jsonArray(str) {
        // 处理剥去两个外层[]的对象
        let curr = 0
        // 指针
        const res = []
        while(curr < str.length) {
            const start = curr
            let stack = 0
            while(curr < str.length && (str[curr] !== ',' || stack !== 0)) {
                if(str[curr] === '{' || str[curr] === '[') {
                    stack++
                }
                if(str[curr] === '}' || str[curr] === ']') {
                    stack--
                }
                curr++
            }
            const val = jsonParser(str.slice(start, curr++))
            res.push(val)
        }
        return res
    }
}

实现深拷贝

// 通过一个map来存储循环引用
const map = new WeakMap()
//这里用WeakMap的原因是防止内存泄漏
function deepClone(target) {
    if(typeof target === 'object') {
        // 如果是克隆目标是对象则要进入递归
        // 再判断是否为数组
        const cloneTarget = Array.isArray(target) ? [] : {}
        if(map.get(target)) {
            return map.get(target)
        }
        //存储一定要放在遍历之前，否则爆栈 
        map.set(target, cloneTarget)
        for(const key in target) {
            cloneTarget[key] = deepClone(target[key])
        }
    }
    return target
}

function myNew(f, ...args) {
    //以f的原型属性为原型创建一个新对象
    const res = Object.create(f.prototype)
    f.call(res, ...args)
    return res 
}
